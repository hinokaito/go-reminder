package main

import "fmt"

const (
	heightLowerLimit = 170
	heightUpperLimit = 200
)

func main() {
	var humanHeight int

	humanHeight = 175
	fmt.Printf("審査の結果、%vでした。\n", checkWarrior(humanHeight))

	humanHeight = 160
	fmt.Printf("審査の結果、%vでした。\n", checkWarrior(humanHeight))

	humanHeight = 201
	fmt.Printf("審査の結果、%vでした。\n", checkWarrior(humanHeight))
}

func checkWarrior(height int) bool {
	if height >= heightLowerLimit && height <= heightUpperLimit {
		return true
	}
	return false
}
